SQL — Criar tabelas (rodar 1x)
-- DuLeoStore CRM · Setup SQL (rodar 1x)
-- Se as tabelas já existem, rode só o bloco ALTER TABLE abaixo

create table if not exists public.dls_leads (
  id text primary key,
  user_id uuid references auth.users(id) on delete cascade,
  nome text, telefone text, tipo text, produto text,
  origem text, status text, valor numeric, obs text,
  data text, ultima_att text,
  created_at timestamptz default now()
);
alter table public.dls_leads enable row level security;
drop policy if exists "dls_leads_own" on public.dls_leads;
create policy "dls_leads_own" on public.dls_leads for all using (auth.uid() = user_id);

create table if not exists public.dls_contatos (
  id text primary key,
